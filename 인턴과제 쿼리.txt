DROP DATABASE IF EXISTS phpboard;
CREATE DATABASE phpboard;
USE phpboard;
CREATE TABLE article (
    id INT UNSIGNED NOT NULL AUTO_INCREMENT,
    PRIMARY KEY(id),
    createDate DATETIME NOT NULL,
    modifyDate DATETIME,
    `title` VARCHAR(100) NOT NULL,
    `body` TEXT NOT NULL
);
SHOW TABLES;
DESC article;
